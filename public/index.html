createChatBtn.onclick = async () => { const phone = newChatPhone.value.trim(); if (!phone) { alert('Введите телефон'); return; } const token = getToken(); if (!token) { alert('Вы не авторизованы'); return; } try { // Сначала ищем пользователя по телефону const res = await fetch(`/api/user/by-phone/${phone}`, { headers: { 'Authorization': `Bearer ${token}` } }); if (!res.ok) { if (res.status === 404) { alert('Пользователь с таким телефоном не найден'); } else { alert('Ошибка при поиске пользователя'); } return; } const user = await res.json(); if (!user || !user.id) { alert('Пользователь не найден'); return; } // Создаём чат const chatRes = await fetch('/api/chats/direct', { method: 'POST', headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, body: JSON.stringify({ userId2: user.id }) }); const chat = await chatRes.json(); if (chatRes.ok && chat.id) { // Обновляем список чатов и открываем новый await loadChats(); await openChat(chat.id); newChatPhone.value = ''; } else { alert('Ошибка при создании чата: ' + (chat.error || 'неизвестная ошибка')); } } catch (error) { console.error('Ошибка:', error); alert('Ошибка соединения с сервером'); } };
